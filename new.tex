\def\blockbase#1#2#3#4#5{
  % \blockbase w h a b z
  %
  %  (0,0)___a   ________________(w,0)
  %      |   \__/                |
  %      |      (a+d+l,b)        |
  %      |  z__b   ______________|
  % (0,-h)     \__/              (w,-h)
  %               (b+d+l,b-#2)
  %
  \def\b{-0.1} % height of \ (part of \_/ )
  \def\d{0.1}  % width  of \ (part of \_/ )
  \def\l{0.25} % width  of _ (part of \_/ )
  \psline(!0 #2 neg) % (0,-h)
    (!0 0)           % (0,0)
    (!#3 0)                                         % (a,0)
    (!#3 \d\space add \b)                           % (a+d,b)
    (!#3 \d\space add \l\space add \b)              % (a+d+l,b)
    (!#3 \d\space dup add add \l\space add 0)       % (a+2d+l,0)
    (!#1 0)          % (w,0)
    (!#1 #2 neg)     % (w,-h)
    (!#4 \d\space dup add add \l\space add #2 neg)  % (b+2d+l,-h)
    (!#4 \d\space add \l\space add \b\space #2 sub) % (b+d+l,b-h)
    (!#4 \d\space add \b\space #2 sub)              % (b+d,b-h)
    (!#4 #2 neg)                                    % (b,-h)
    (!#5 #2 neg)     % (z,-h)
}
